Mathematics 101

Fill up to 5 fields with a number to begin some basic calculations. Calculate will show you the smallest number, the largest number, the average of the numbers, the sum total of the numbers, and the multiplied value of the numbers. A field containing a letter is excluded. 

//script to pull data and calc it
//the calc btn
$('#BM-btn-1').click(function () {
    //establish variables and calculatable array
    var count = 0; var sum = 0; var mult = 0; var least = 0; var most = 0; var ave = 0;
    let calcVals = new Array();
    //run a loop that checks each field to see it there is something in it,
    for (i = 1; i <= 5; i++) {
        var str = '#UTI-' + i;
        if ($(str).val() != "" && !isNaN(Number($(str).val()))) { //a field is excluded if it is blank or NaN
            calcVals.push(Number($(str).val())); //push the non-excluded fields to a new array
            if (calcVals.length == 1) { //each loop pass will assign a new highest and lowest
                least = calcVals[count];
                most = calcVals[count];
                mult = 1;
            }
            if (calcVals[count] > most) { //assigning highest
                most = calcVals[count];
            }
            else if (calcVals[count] < least) { //assigning lowest
                least = calcVals[count];
            }

            sum = sum += calcVals[count]; //calculate the total sum by adding each array item value together
            mult = mult * calcVals[count]; //multiply all non-excluded fields together
            ave = (sum / calcVals.length); //average the fields using the sum
            count++
        }
    }
    //output the results
    var out1 = "The smallest of your #'s is " + least + ".";
    var out2 = "The largest of your #'s is " + most + ".";
    var out3 = "The average of your #'s is " + ave + ".";
    var out4 = "The sum total of your #'s is " + sum + ".";
    var out5 = "The product of your #'s is " + mult + ".";
    $('#result1').text(out1);
    $('#result2').text(out2);
    $('#result3').text(out3);
    $('#result4').text(out4);
    $('#result5').text(out5);
});
